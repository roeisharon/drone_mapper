# Mission configuration
# Scenario 4: Solid internal obstacle (produces -1) + height boundary (produces -2)
#
# SOURCE OF -1 (NotMapped / unreachable interior):
#   A solid box fills x=80-120, y=80-120, z=0-100.  The drone can detect and
#   map the 6 outer faces, but the interior cells (x=85-115, y=85-115, z=5-95)
#   are hidden behind solid material.  No lidar beam can penetrate solid cells,
#   so those ~931 cells are never reached → NotMapped (-1) in output.
#
# SOURCE OF -2 (NotReachable / outside mapping boundaries):
#   The mission height range is z=5-95.  The floor (z=0) and ceiling (z=100)
#   are outside the drone's mapping boundaries.  The drone's MapBuilder drops
#   any attempt to record those cells, and Get() returns NotReachable (-2).
#   All 41×41×2 = 3 362 floor/ceiling cells appear as -2 in the output.

# Mapping boundary polygon (x,y in cm): full room rectangle
boundary_polygon = (0,0),(200,0),(200,200),(0,200)

# Height range (cm): deliberately excludes floor (z=0) and ceiling (z=100)
min_height_cm = 5
max_height_cm = 95

# Drone starting position (cm): lower-left area, clear of the solid box
start_x_cm      = 50
start_y_cm      = 50
start_height_cm = 50

# Output resolution (decimal places after dot)
output_resolution_xy = 0
output_resolution_h  = 0

# Grid cell size for output map (cm); must match the input map resolution
map_resolution_cm = 5
